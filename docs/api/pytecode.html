pytecode

Python library for parsing, inspecting, manipulating, and emitting JVM class files.

Provides four top-level entry points:

  • ClassReader — parse .class bytes into a ClassFile tree.
  • ClassWriter — serialize a ClassFile tree back to .class bytes.
  • ClassModel — mutable editing model with symbolic references and label-aware code editing.
  • JarFile — read, mutate, and rewrite JAR archives.

Additional submodules expose transforms, descriptors, analysis, hierarchy resolution, validation, operands, labels, debug-info helpers, and the underlying data types. See pytecode.transforms, pytecode.analysis, pytecode.verify, and the other documented submodules for details.

 92        elif cp_type is constant_pool.ConstantPoolInfoType.STRING:

 95            cp_info = constant_pool.MethodTypeInfo(index, offset, tag, self.read_u2())

102        elif cp_type is constant_pool.ConstantPoolInfoType.METHOD_REF:
103            cp_info = constant_pool.MethodrefInfo(index, offset, tag, self.read_u2(), self.read_u2())
104        elif cp_type is constant_pool.ConstantPoolInfoType.INTERFACE_METHOD_REF:
105            cp_info = constant_pool.InterfaceMethodrefInfo(index, offset, tag, self.read_u2(), self.read_u2())
106        elif cp_type is constant_pool.ConstantPoolInfoType.NAME_AND_TYPE:

108        elif cp_type is constant_pool.ConstantPoolInfoType.DYNAMIC:
109            cp_info = constant_pool.DynamicInfo(index, offset, tag, self.read_u2(), self.read_u2())
110        elif cp_type is constant_pool.ConstantPoolInfoType.INVOKE_DYNAMIC:
111            cp_info = constant_pool.InvokeDynamicInfo(index, offset, tag, self.read_u2(), self.read_u2())
112        elif cp_type is constant_pool.ConstantPoolInfoType.INTEGER:
113            cp_info = constant_pool.IntegerInfo(index, offset, tag, self.read_u4())
114        elif cp_type is constant_pool.ConstantPoolInfoType.FLOAT:
115            cp_info = constant_pool.FloatInfo(index, offset, tag, self.read_u4())
116        elif cp_type is constant_pool.ConstantPoolInfoType.LONG:
117            cp_info = constant_pool.LongInfo(index, offset, tag, self.read_u4(), self.read_u4())
118            index_extra = 1
119        elif cp_type is constant_pool.ConstantPoolInfoType.DOUBLE:
120            cp_info = constant_pool.DoubleInfo(index, offset, tag, self.read_u4(), self.read_u4())
121            index_extra = 1
122        elif cp_type is constant_pool.ConstantPoolInfoType.UTF8:
123            length = self.read_u2()
124            str_bytes = self.read_bytes(length)
125            cp_info = constant_pool.Utf8Info(index, offset, tag, length, str_bytes)
126        elif cp_type is constant_pool.ConstantPoolInfoType.METHOD_HANDLE:
127            cp_info = constant_pool.MethodHandleInfo(index, offset, tag, self.read_u1(), self.read_u2())
128        else:
129            raise ValueError(f"Unknown ConstantPoolInfoType: {cp_type}")
130        return cp_info, index_extra

132    def read_align_bytes(self, current_offset: int) -> bytes:
133        """Read and discard padding bytes to reach 4-byte alignment.
134
135        Used by ``tableswitch`` and ``lookupswitch`` instructions whose
136        operands must be 4-byte aligned (JVMS §6.5).
137
138        Args:
139            current_offset: Current bytecode offset within the method body.
140
141        Returns:
142            The consumed padding bytes (0–3 bytes).
143        """
144        align_bytes = (4 - current_offset % 4) % 4
145        return self.read_bytes(align_bytes)

147    def read_instruction(self, current_method_offset: int) -> instructions.InsnInfo:
148        """Read a single JVM bytecode instruction (JVMS §6.5).
149
150        Args:
151            current_method_offset: Byte offset of this instruction relative to
152                the start of the method's ``Code`` attribute bytecode array.
153
154        Returns:
155            The decoded instruction info object.
156
157        Raises:
158            Exception: If the opcode or its ``wide`` variant is invalid.
159        """
160        opcode = self.read_u1()
161        inst_type = instructions.InsnInfoType(opcode)
162        instinfo = inst_type.instinfo
163        if instinfo is instructions.LocalIndex:
164            return instructions.LocalIndex(inst_type, current_method_offset, self.read_u1())
165        elif instinfo is instructions.ConstPoolIndex:
166            return instructions.ConstPoolIndex(inst_type, current_method_offset, self.read_u2())
167        elif instinfo is instructions.ByteValue:
168            return instructions.ByteValue(inst_type, current_method_offset, self.read_i1())
169        elif instinfo is instructions.ShortValue:
170            return instructions.ShortValue(inst_type, current_method_offset, self.read_i2())
171        elif instinfo is instructions.Branch:
172            return instructions.Branch(inst_type, current_method_offset, self.read_i2())
173        elif instinfo is instructions.BranchW:
174            return instructions.BranchW(inst_type, current_method_offset, self.read_i4())
175        elif instinfo is instructions.IInc:
176            index, value = self.read_u1(), self.read_i1()
177            return instructions.IInc(inst_type, current_method_offset, index, value)
178        elif instinfo is instructions.InvokeDynamic:
179            index, unused = self.read_u2(), self.read_bytes(2)
180            return instructions.InvokeDynamic(inst_type, current_method_offset, index, unused)
181        elif instinfo is instructions.InvokeInterface:
182            index, count, unused = self.read_u2(), self.read_u1(), self.read_bytes(1)
183            return instructions.InvokeInterface(inst_type, current_method_offset, index, count, unused)
184        elif instinfo is instructions.MultiANewArray:
185            index, dimensions = self.read_u2(), self.read_u1()
186            return instructions.MultiANewArray(inst_type, current_method_offset, index, dimensions)
187        elif instinfo is instructions.NewArray:
188            atype = instructions.ArrayType(self.read_u1())
189            return instructions.NewArray(inst_type, current_method_offset, atype)
190        elif instinfo is instructions.LookupSwitch:
191            self.read_align_bytes(current_method_offset + 1)
192            default, npairs = self.read_i4(), self.read_u4()
193            pairs = [instructions.MatchOffsetPair(self.read_i4(), self.read_i4()) for _ in range(npairs)]
194            return instructions.LookupSwitch(inst_type, current_method_offset, default, npairs, pairs)
195        elif instinfo is instructions.TableSwitch:
196            self.read_align_bytes(current_method_offset + 1)
197            default, low, high = self.read_i4(), self.read_i4(), self.read_i4()
198            offsets = [self.read_i4() for _ in range(high - low + 1)]
199            return instructions.TableSwitch(inst_type, current_method_offset, default, low, high, offsets)
200        elif inst_type is instructions.InsnInfoType.WIDE:
201            wide_opcode = self.read_u1()
202            wide_inst_type = instructions.InsnInfoType(opcode + wide_opcode)
203            if wide_inst_type.instinfo is instructions.LocalIndexW:
204                return instructions.LocalIndexW(wide_inst_type, current_method_offset, self.read_u2())
205            elif wide_inst_type.instinfo is instructions.IIncW:
206                index, value = self.read_u2(), self.read_i2()
207                return instructions.IIncW(wide_inst_type, current_method_offset, index, value)
208        elif instinfo is instructions.InsnInfo:
209            return instructions.InsnInfo(inst_type, current_method_offset)
210
211        raise Exception(f"Invalid InstInfoType: {inst_type.name} {inst_type.instinfo}")
212
213    def read_code_bytes(self, code_length: int) -> list[instructions.InsnInfo]:
214        """Read the full bytecode array of a ``Code`` attribute (JVMS §4.7.3).
215
216        Args:
217            code_length: Number of bytes in the bytecode array.
218
219        Returns:
220            Ordered list of decoded instructions.
221        """
222        start_method_offset = self.offset
223        results: list[instructions.InsnInfo] = []
224        while (current_method_offset := self.offset - start_method_offset) < code_length:
225            insn = self.read_instruction(current_method_offset)
226            results.append(insn)
227        return results

261    def read_element_value_info(self) -> attributes.ElementValueInfo:
262        """Read an ``element_value`` structure from an annotation (JVMS §4.7.16.1).
263
264        Returns:
265            The decoded element-value info.
266
267        Raises:
268            ValueError: If the element-value tag character is unrecognised.
269        """
270        tag = self.read_u1().to_bytes(1, "big").decode("ascii")
271
272        match tag:
273            case x if x in ("B", "C", "D", "F", "I", "J", "S", "Z", "s"):
274                return attributes.ElementValueInfo(tag, attributes.ConstValueInfo(self.read_u2()))
275            case "e":
276                return attributes.ElementValueInfo(
277                    tag,
278                    attributes.EnumConstantValueInfo(self.read_u2(), self.read_u2()),
279                )
280            case "c":
281                return attributes.ElementValueInfo(tag, attributes.ClassInfoValueInfo(self.read_u2()))
282            case "@":
283                return attributes.ElementValueInfo(tag, self.read_annotation_info())
284            case "[":
285                num_values = self.read_u2()
286                values = [self.read_element_value_info() for _ in range(num_values)]
287                return attributes.ElementValueInfo(tag, attributes.ArrayValueInfo(num_values, values))
288            case _:
289                raise ValueError(f"Unknown element value tag: {tag}")

379    def read_attribute(self) -> attributes.AttributeInfo:
380        """Read a single ``attribute_info`` structure (JVMS §4.7).
381
382        Recognised attribute names are decoded into their specific subtypes;
383        unknown attributes are returned as :class:`~pytecode.attributes.UnimplementedAttr`.
384
385        Returns:
386            The decoded attribute info.
387
388        Raises:
389            ValueError: If the attribute name index does not reference a
390                ``CONSTANT_Utf8_info`` entry.
391        """
392        name_index, length = self.read_u2(), self.read_u4()
393
394        name_cp = self.constant_pool[name_index]
395        if not isinstance(name_cp, constant_pool.Utf8Info):
396            raise ValueError(f"name_index({name_index}) should be Utf8Info, not {type(name_cp)}")
397
398        name = decode_modified_utf8(name_cp.str_bytes)
399        attr_type = attributes.AttributeInfoType(name)
400
401        if attr_type is attributes.AttributeInfoType.SYNTHETIC:
402            return attributes.SyntheticAttr(name_index, length)
403
404        elif attr_type is attributes.AttributeInfoType.DEPRECATED:
405            return attributes.DeprecatedAttr(name_index, length)
406
407        elif attr_type is attributes.AttributeInfoType.CONSTANT_VALUE:
408            return attributes.ConstantValueAttr(name_index, length, self.read_u2())
409
410        elif attr_type is attributes.AttributeInfoType.SIGNATURE:
411            return attributes.SignatureAttr(name_index, length, self.read_u2())
412
413        elif attr_type is attributes.AttributeInfoType.SOURCE_FILE:
414            return attributes.SourceFileAttr(name_index, length, self.read_u2())
415
416        elif attr_type is attributes.AttributeInfoType.MODULE_MAIN_CLASS:
417            return attributes.ModuleMainClassAttr(name_index, length, self.read_u2())
418
419        elif attr_type is attributes.AttributeInfoType.NEST_HOST:
420            return attributes.NestHostAttr(name_index, length, self.read_u2())
421
422        elif attr_type is attributes.AttributeInfoType.CODE:
423            max_stack, max_locals = self.read_u2(), self.read_u2()
424            code_length = self.read_u4()
425            code = self.read_code_bytes(code_length)
426            exception_table_length = self.read_u2()
427            exception_table = [
428                attributes.ExceptionInfo(self.read_u2(), self.read_u2(), self.read_u2(), self.read_u2())
429                for _ in range(exception_table_length)
430            ]
431            attributes_count = self.read_u2()
432            attributes_list = [self.read_attribute() for _ in range(attributes_count)]
433            return attributes.CodeAttr(
434                name_index,
435                length,
436                max_stack,
437                max_locals,
438                code_length,
439                code,
440                exception_table_length,
441                exception_table,
442                attributes_count,
443                attributes_list,
444            )
445
446        elif attr_type is attributes.AttributeInfoType.STACK_MAP_TABLE:
447            number_of_entries = self.read_u2()
448            entries: list[attributes.StackMapFrameInfo] = []
449            for _ in range(number_of_entries):
450                frame_type = self.read_u1()
451
452                match frame_type:
453                    case x if x in range(0, 64):
454                        entries.append(attributes.SameFrameInfo(frame_type))
455                    case x if x in range(64, 128):
456                        entries.append(
457                            attributes.SameLocals1StackItemFrameInfo(frame_type, self.read_verification_type_info())
458                        )
459                    case 247:
460                        entries.append(
461                            attributes.SameLocals1StackItemFrameExtendedInfo(
462                                frame_type,
463                                self.read_u2(),
464                                self.read_verification_type_info(),
465                            )
466                        )
467                    case x if x in range(248, 251):
468                        entries.append(attributes.ChopFrameInfo(frame_type, self.read_u2()))
469                    case 251:
470                        entries.append(attributes.SameFrameExtendedInfo(frame_type, self.read_u2()))
471                    case x if x in range(252, 255):
472                        offset_delta = self.read_u2()
473                        verification_type_infos = [self.read_verification_type_info() for __ in range(frame_type - 251)]
474                        entries.append(attributes.AppendFrameInfo(frame_type, offset_delta, verification_type_infos))
475                    case 255:
476                        offset_delta = self.read_u2()
477                        number_of_locals = self.read_u2()
478                        locals = [self.read_verification_type_info() for __ in range(number_of_locals)]
479                        number_of_stack_items = self.read_u2()
480                        stack = [self.read_verification_type_info() for __ in range(number_of_stack_items)]
481                        entries.append(
482                            attributes.FullFrameInfo(
483                                frame_type,
484                                offset_delta,
485                                number_of_locals,
486                                locals,
487                                number_of_stack_items,
488                                stack,
489                            )
490                        )
491                    case _:
492                        raise ValueError(f"Unknown stack map frame type: {frame_type}")
493
494            return attributes.StackMapTableAttr(name_index, length, number_of_entries, entries)
495
496        elif attr_type is attributes.AttributeInfoType.EXCEPTIONS:
497            number_of_exceptions = self.read_u2()
498            exception_index_table = [self.read_u2() for _ in range(number_of_exceptions)]
499            return attributes.ExceptionsAttr(name_index, length, number_of_exceptions, exception_index_table)
500
501        elif attr_type is attributes.AttributeInfoType.INNER_CLASSES:
502            number_of_classes = self.read_u2()
503            classes = [
504                attributes.InnerClassInfo(
505                    self.read_u2(),
506                    self.read_u2(),
507                    self.read_u2(),
508                    constants.NestedClassAccessFlag(self.read_u2()),
509                )
510                for _ in range(number_of_classes)
511            ]
512            return attributes.InnerClassesAttr(name_index, length, number_of_classes, classes)
513
514        elif attr_type is attributes.AttributeInfoType.ENCLOSING_METHOD:
515            return attributes.EnclosingMethodAttr(name_index, length, self.read_u2(), self.read_u2())
516
517        elif attr_type is attributes.AttributeInfoType.SOURCE_DEBUG_EXTENSION:
518            return attributes.SourceDebugExtensionAttr(name_index, length, self.read_bytes(length).decode("utf-8"))
519
520        elif attr_type is attributes.AttributeInfoType.LINE_NUMBER_TABLE:
521            line_number_table_length = self.read_u2()
522            line_number_table = [
523                attributes.LineNumberInfo(self.read_u2(), self.read_u2()) for _ in range(line_number_table_length)
524            ]
525            return attributes.LineNumberTableAttr(name_index, length, line_number_table_length, line_number_table)
526
527        elif attr_type is attributes.AttributeInfoType.LOCAL_VARIABLE_TABLE:
528            local_variable_table_length = self.read_u2()
529            local_variable_table = [
530                attributes.LocalVariableInfo(
531                    self.read_u2(),
532                    self.read_u2(),
533                    self.read_u2(),
534                    self.read_u2(),
535                    self.read_u2(),
536                )
537                for _ in range(local_variable_table_length)
538            ]
539            return attributes.LocalVariableTableAttr(
540                name_index, length, local_variable_table_length, local_variable_table
541            )
542
543        elif attr_type is attributes.AttributeInfoType.LOCAL_VARIABLE_TYPE_TABLE:
544            local_variable_type_table_length = self.read_u2()
545            local_variable_type_table = [
546                attributes.LocalVariableTypeInfo(
547                    self.read_u2(),
548                    self.read_u2(),
549                    self.read_u2(),
550                    self.read_u2(),
551                    self.read_u2(),
552                )
553                for _ in range(local_variable_type_table_length)
554            ]
555            return attributes.LocalVariableTypeTableAttr(
556                name_index, length, local_variable_type_table_length, local_variable_type_table
557            )
558
559        elif attr_type is attributes.AttributeInfoType.RUNTIME_VISIBLE_ANNOTATIONS:
560            num_annotations = self.read_u2()
561            annotation_list = [self.read_annotation_info() for _ in range(num_annotations)]
562            return attributes.RuntimeVisibleAnnotationsAttr(name_index, length, num_annotations, annotation_list)
563
564        elif attr_type is attributes.AttributeInfoType.RUNTIME_INVISIBLE_ANNOTATIONS:
565            num_annotations = self.read_u2()
566            annotation_list = [self.read_annotation_info() for _ in range(num_annotations)]
567            return attributes.RuntimeInvisibleAnnotationsAttr(name_index, length, num_annotations, annotation_list)
568
569        elif attr_type is attributes.AttributeInfoType.RUNTIME_VISIBLE_PARAMETER_ANNOTATIONS:
570            num_parameters = self.read_u1()
571            parameter_annotations: list[attributes.ParameterAnnotationInfo] = []
572            for _ in range(num_parameters):
573                num_annotations = self.read_u2()
574                annotation_list = [self.read_annotation_info() for _ in range(num_annotations)]
575                parameter_annotations.append(attributes.ParameterAnnotationInfo(num_annotations, annotation_list))
576            return attributes.RuntimeVisibleParameterAnnotationsAttr(
577                name_index, length, num_parameters, parameter_annotations
578            )
579
580        elif attr_type is attributes.AttributeInfoType.RUNTIME_INVISIBLE_PARAMETER_ANNOTATIONS:
581            num_parameters = self.read_u1()
582            parameter_annotations_list: list[attributes.ParameterAnnotationInfo] = []
583            for _ in range(num_parameters):
584                num_annotations = self.read_u2()
585                annotation_list = [self.read_annotation_info() for _ in range(num_annotations)]
586                parameter_annotations_list.append(attributes.ParameterAnnotationInfo(num_annotations, annotation_list))
587            return attributes.RuntimeInvisibleParameterAnnotationsAttr(
588                name_index, length, num_parameters, parameter_annotations_list
589            )
590
591        elif attr_type is attributes.AttributeInfoType.RUNTIME_VISIBLE_TYPE_ANNOTATIONS:
592            num_annotations = self.read_u2()
593            type_annotation_list = [self.read_type_annotation_info() for _ in range(num_annotations)]
594            return attributes.RuntimeVisibleTypeAnnotationsAttr(
595                name_index, length, num_annotations, type_annotation_list
596            )
597
598        elif attr_type is attributes.AttributeInfoType.RUNTIME_INVISIBLE_TYPE_ANNOTATIONS:
599            num_annotations = self.read_u2()
600            type_annotation_list = [self.read_type_annotation_info() for _ in range(num_annotations)]
601            return attributes.RuntimeInvisibleTypeAnnotationsAttr(
602                name_index, length, num_annotations, type_annotation_list
603            )
604
605        elif attr_type is attributes.AttributeInfoType.ANNOTATION_DEFAULT:
606            return attributes.AnnotationDefaultAttr(name_index, length, self.read_element_value_info())
607
608        elif attr_type is attributes.AttributeInfoType.BOOTSTRAP_METHODS:
609            num_bootstrap_methods = self.read_u2()
610            bootstrap_methods: list[attributes.BootstrapMethodInfo] = []
611            for _ in range(num_bootstrap_methods):
612                bootstrap_method_ref = self.read_u2()
613                num_bootstrap_arguments = self.read_u2()
614                bootstrap_arguments = [self.read_u2() for __ in range(num_bootstrap_arguments)]
615                bootstrap_methods.append(
616                    attributes.BootstrapMethodInfo(
617                        bootstrap_method_ref,
618                        num_bootstrap_arguments,
619                        bootstrap_arguments,
620                    )
621                )
622            return attributes.BootstrapMethodsAttr(name_index, length, num_bootstrap_methods, bootstrap_methods)
623
624        elif attr_type is attributes.AttributeInfoType.METHOD_PARAMETERS:
625            parameters_count = self.read_u1()
626            parameters = [
627                attributes.MethodParameterInfo(self.read_u2(), constants.MethodParameterAccessFlag(self.read_u2()))
628                for _ in range(parameters_count)
629            ]
630            return attributes.MethodParametersAttr(name_index, length, parameters_count, parameters)
631
632        elif attr_type is attributes.AttributeInfoType.MODULE:
633            module_name_index = self.read_u2()
634            module_flags = constants.ModuleAccessFlag(self.read_u2())
635            module_version_index = self.read_u2()
636
637            requires_count = self.read_u2()
638            requires = [
639                attributes.RequiresInfo(
640                    self.read_u2(),
641                    constants.ModuleRequiresAccessFlag(self.read_u2()),
642                    self.read_u2(),
643                )
644                for _ in range(requires_count)
645            ]
646
647            exports_count = self.read_u2()
